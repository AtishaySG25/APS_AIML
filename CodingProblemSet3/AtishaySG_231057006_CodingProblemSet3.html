import matplotlib.pyplot as plt
import numpy as np
from scipy.stats import norm, gamma, lognorm, skewnorm

# Read data from the text file, skipping the first row
data_A = np.genfromtxt('personKeyTimingA.txt', delimiter=',', names=['time', 'key'], dtype=None, encoding=None, skip_header=1)
data_B = np.genfromtxt('personKeyTimingB.txt', delimiter=',', names=['time', 'key'], dtype=None, encoding=None, skip_header=1)

# Convert the 'time' column to numeric
data_A['time'] = data_A['time'].astype(float)
data_B['time'] = data_B['time'].astype(float)

# Calculate keystroke times
times_A = np.diff(data_A['time'])
times_B = np.diff(data_B['time'])

# Fit a normal distribution to the data
params_A = norm.fit(times_A)
params_B = norm.fit(times_B)

# Plot the histograms
plt.hist(times_A, bins=50, density=True, alpha=0.5, label='User A')
plt.hist(times_B, bins=50, density=True, alpha=0.5, label='User B')

# Plot the PDFs of the fitted distributions
xmin, xmax = plt.xlim()
x_A = np.linspace(xmin, xmax, 100)
pdf_A = norm.pdf(x_A, *params_A)
plt.plot(x_A, pdf_A, 'k-', label='User A (Normal)')

x_B = np.linspace(xmin, xmax, 100)
pdf_B = norm.pdf(x_B, *params_B)
plt.plot(x_B, pdf_B, 'k--', label='User B (Normal)')

# Add labels and title
plt.xlabel('Keystroke Time (milliseconds)')
plt.ylabel('Density')
plt.title('Density Histograms and Fitted Distributions for Users A and B')

# Add a legend
plt.legend()

# Show the plot
plt.show()

# Display parameters of the fitted distributions
print("User A (Normal) Distribution Parameters:", params_A)
print("User B (Normal) Distribution Parameters:", params_B)

User A (Normal) Distribution Parameters: (7.406662134602312, 1.9958138908902976)
User B (Normal) Distribution Parameters: (8.029843643779742, 1.9195689565503107)
